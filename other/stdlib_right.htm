WoG functions, StdLib

Expansion to WoG functions, realising some common tasks without using ERM receivers.


StdLib.Lod
Functions allowing to load/unload Custom Lods.
StdLib.Lod = {
	-- 
	Temp = function() ... end,
	Stored = function() ... end,
	Perm = function() ... end,

	Unload = function(ind) ... end,
	-- ind - index of LOD file to be unloaded
	-- return -1 if there was error
	--         0 if everything went smoothly
	
	Load = function(location, name, mode) ... end
	-- location - modifier for location, like in UN:J8
	-- name - name of LOD file
	-- mode - mode/kind
	-- return index of loaded lod, -1 if error occured
}


StdLib.Objects
Now you can iterate through/access objects without claiming ERM variables.
StdLib.Objects = {
	Calculate = function(t,st) ... end,
	-- t - type
	-- st - subtype
	-- return: number of objects of given type/subtype
	
	Iterate = function(t,st,func,args) ... end,
	-- t - type
	-- st - subtype
	-- func - function(x,y,z,args) to be called in loop
	-- args - arguments passed as "args" to "func"
	-- return -1 if error occured
	
	Find = function(t,st,ind) ... end
	-- t - type
	-- st - subtype
	-- ind - number of object (1..num)
	-- return x,y,z value
	-- if error occured, x,y,z are filled with 'nil'
}


StdLib.Universal
StdLib.Universal = {
	DoesFileExist = function(name, location) ... end
	-- name - name of file
	-- location - modifier for location, like in UN:J8
	-- check if file exists
	-- 0 means error or doesnt exists
	-- 1 means file exists
}
